# Deploy nginx ingress controller
https://kubernetes.github.io/ingress-nginx/deploy/

# Create secret
kubectl create secret tls nginx-ingress-tls --key .\deploy\certs\cert.key --cert .\deploy\certs\fullchain.crt

# Mandatory stuff
kubectl apply -f https://raw.githubusercontent.com/kubernetes/ingress-nginx/master/deploy/static/mandatory.yaml --watch-namespace test-reactive-analytics

# GCP Provider Specific
kubectl apply -f https://raw.githubusercontent.com/kubernetes/ingress-nginx/master/deploy/static/provider/cloud-generic.yaml
